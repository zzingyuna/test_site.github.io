	var time = 30;
	var finish;

	// 게임 시작
	function Start(){
		time = 30;
		$("#time").text(time);
		finish = setInterval(CountTime, 1000);
		$("img").hide();
		$("a.startbtn").hide();
		openitem = "";
		$("td").bind('click', function(e){
			e.preventDefault();
			ClickEvent(this.lastChild);
		});
	}

	function CountTime() {
		time = time -1;
		$("#time").text(time);
		if(time < 1){
			alert("GAME OVER!!");
			$("td").off("click");
			clearInterval(finish);
			$("a.startbtn").show();
		}
	}

	function ClickEvent(nowitem){
		if(openitem == undefined || openitem == ""){
			openitem = nowitem.name;
			$(nowitem).show();
			$(nowitem.parentElement).off('click');
		}else if(openitem==nowitem.name){
			$(nowitem).show();
			$("img[name="+openitem+"]").parent().off('click');
			$(nowitem.parentElement).off('click');
			openitem = "";
		}else{
			$(nowitem).show();
			$("td").off('click');
			setTimeout(function(){
				//alert("틀립니다.");
				$(nowitem).hide();
				$("img[name="+openitem+"]").hide();
				openitem = "";
				$("td").has("img:hidden").bind('click', function(e){
					ClickEvent(this.lastChild);
				});
			}, 500);

		}

			$("td").off('click');
